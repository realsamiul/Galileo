## The Architecture Now

```
User (Messenger/WhatsApp)
    → API Gateway 
    → Lambda: proofsheet-webhook (message handler + Bedrock call)
    → If [DOCUMENT_READY]: 
        → Lambda: proofsheet-renderer (container image, WeasyPrint, Bengali fonts)
        → PDF bytes back
        → Send PDF as file attachment
    → Response back to user
```

We're splitting into **two Lambdas**:
1. **proofsheet-webhook** — handles messages, calls Bedrock, manages conversation
2. **proofsheet-renderer** — takes HTML, produces PDF (container image with fonts)

Why split? The renderer needs a container image (for WeasyPrint + Bengali fonts). The webhook Lambda stays lightweight and fast. The renderer is invoked synchronously only when a document is ready.

def lambda_handler(event, context):
    print(f"Event: {json.dumps(event)}")
    method = event.get('requestContext', {}).get('http', {}).get('method', 'POST')

    if method == 'GET':
        return handle_verification(event)

    try:
        body = json.loads(event.get('body', '{}'))
    except json.JSONDecodeError:
        return {'statusCode': 200}

    obj = body.get('object', '')

    if obj == 'whatsapp_business_account':
        handle_whatsapp(body)
    elif obj == 'page':
        handle_messenger(body)

    return {'statusCode': 200}


def handle_verification(event):
    params = event.get('queryStringParameters', {}) or {}
    mode = params.get('hub.mode', '')
    token = params.get('hub.verify_token', '')
    challenge = params.get('hub.challenge', '')

    secrets = get_secrets()
    if mode == 'subscribe' and token == secrets['verify_token']:
        return {'statusCode': 200, 'headers': {'Content-Type': 'text/plain'}, 'body': challenge}
    return {'statusCode': 403, 'body': 'Forbidden'}

def handle_whatsapp(body):
    for entry in body.get('entry', []):
        for change in entry.get('changes', []):
            value = change.get('value', {})
            if 'messages' not in value:
                continue
            for message in value['messages']:
                sender = message['from']
                msg_type = message['type']
                if msg_type == 'text':
                    user_text = message['text']['body']
                elif msg_type == 'interactive':
                    interactive = message.get('interactive', {})
                    user_text = (interactive.get('button_reply', {}).get('title') or
                                interactive.get('list_reply', {}).get('title') or '[interactive]')
                else:
                    send_whatsapp_reply(sender, "এখন শুধু text support করি। Voice ও image শীঘ্রই আসছে! 🙏")
                    return
                process_and_respond(sender, user_text, 'whatsapp')


def handle_messenger(body):
    for entry in body.get('entry', []):
        for messaging_event in entry.get('messaging', []):
            if messaging_event.get('message', {}).get('is_echo'):
                continue
            sender_id = messaging_event['sender']['id']
            if 'message' in messaging_event:
                message = messaging_event['message']
                if 'text' in message:
                    user_text = message['text']
                elif 'attachments' in message:
                    send_messenger_reply(sender_id, "এখন শুধু text support করি। Voice ও image শীঘ্রই আসছে! 🙏")
                    return
                else:
                    return
                process_and_respond(sender_id, user_text, 'messenger')
            elif 'postback' in messaging_event:
                payload = messaging_event['postback'].get('title', messaging_event['postback'].get('payload', ''))
                process_and_respond(sender_id, payload, 'messenger')

def call_bedrock(user_text, history, user_profile, last_doc):
    today = datetime.now(timezone.utc).strftime('%d %B %Y')
    
    # Build user profile section
    if user_profile:
        profile_lines = []
        if user_profile.get('name'): profile_lines.append(f"Name: {user_profile['name']}")
        if user_profile.get('business_name'): profile_lines.append(f"Business: {user_profile['business_name']}")
        if user_profile.get('business_address'): profile_lines.append(f"Address: {user_profile['business_address']}")
        if user_profile.get('phone'): profile_lines.append(f"Phone: {user_profile['phone']}")
        if user_profile.get('email'): profile_lines.append(f"Email: {user_profile['email']}")
        user_profile_section = '\n'.join(profile_lines) if profile_lines else "No profile saved yet."
    else:
        user_profile_section = "No profile saved yet. Learn from conversation."
    
    # Build last document section
    if last_doc:
        last_doc_section = f"Type: {last_doc.get('doc_type', 'unknown')}\nData: {json.dumps(last_doc.get('doc_data', {}), ensure_ascii=False)[:1500]}"
    else:
        last_doc_section = "No previous document in this session."
    
    # Fill template
    system = SYSTEM_PROMPT.replace('{today}', today)\
                          .replace('{user_profile_section}', user_profile_section)\
                          .replace('{last_doc_section}', last_doc_section)
    
    # Build messages
    messages = []
    for msg in history[-14:]:  # Last 14 messages (7 turns)
        messages.append({"role": msg['role'], "content": msg['content']})
    messages.append({"role": "user", "content": user_text})

    try:
        response = bedrock.invoke_model(
            modelId='anthropic.claude-sonnet-4-20250514',
            body=json.dumps({
                "anthropic_version": "bedrock-2023-05-31",
                "max_tokens": 4096,
                "system": system,
                "messages": messages
            })
        )
        result = json.loads(response['body'].read())
        return result['content'][0]['text']
    except Exception as e:
        print(f"Bedrock error: {e}")
        return "দুঃখিত, একটু সমস্যা হয়েছে। আবার চেষ্টা করুন। 🙏"

def handle_document_generation(sender_id, response, channel):
    parts = response.split('[DOCUMENT_READY]')
    conversational_reply = parts[0].strip()
    
    # Clean the JSON part
    json_part = parts[1]
    # Remove [PROFILE_UPDATE] if it's after the doc
    if '[PROFILE_UPDATE]' in json_part:
        json_part = json_part.split('[PROFILE_UPDATE]')[0]
    json_part = json_part.strip().replace('```json', '').replace('```', '').strip()

    try:
        doc_data = json.loads(json_part)
    except json.JSONDecodeError as e:
        print(f"JSON parse error: {e}\nRaw: {json_part[:500]}")
        send_response(channel, sender_id, conversational_reply + "\n\n⚠️ Document generation এ সমস্যা হয়েছে। আবার চেষ্টা করুন।")
        return

    doc_type = doc_data.get('type', 'unknown')
    doc_content = doc_data.get('data', {})
    
    # Send conversational reply immediately
    send_response(channel, sender_id, conversational_reply + "\n\n📄 তৈরি হচ্ছে...")

    # Render HTML
    html_content = render_document(doc_type, doc_content)
    
    # Generate PDF via renderer Lambda
    pdf_bytes = invoke_renderer(html_content, doc_type)
    
    if pdf_bytes:
        # Generate filename
        timestamp_str = datetime.now(timezone.utc).strftime('%Y%m%d_%H%M%S')
        filename = f"{doc_type}_{timestamp_str}.pdf"
        
        # Save to S3
        s3_key = f"{sender_id}/{filename}"
        try:
            s3.put_object(
                Bucket=DOCS_BUCKET,
                Key=s3_key,
                Body=pdf_bytes,
                ContentType='application/pdf'
            )
        except Exception as e:
            print(f"S3 error: {e}")
        
        # Send PDF to user
        send_file_response(channel, sender_id, pdf_bytes, filename)
        
        # Save document record (for editing)
        save_document_record(sender_id, doc_type, doc_content, s3_key)
    else:
        # Fallback: send HTML as file
        html_bytes = html_content.encode('utf-8')
        filename = f"{doc_type}_{datetime.now(timezone.utc).strftime('%Y%m%d_%H%M%S')}.html"
        send_file_response(channel, sender_id, html_bytes, filename)
        send_response(channel, sender_id, "💡 PDF renderer is starting up. Sent as HTML — opens beautifully in any browser!")
        save_document_record(sender_id, doc_type, doc_content, None)


def invoke_renderer(html_content, doc_type):
    """Call the renderer Lambda to convert HTML → PDF"""
    try:
        response = lambda_client.invoke(
            FunctionName='proofsheet-renderer',
            InvocationType='RequestResponse',
            Payload=json.dumps({
                'html': html_content,
                'doc_type': doc_type,
                'options': {
                    'page_size': 'A4',
                    'margin': '15mm'
                }
            })
        )
        result = json.loads(response['Payload'].read())
        
        if result.get('statusCode') == 200:
            import base64
            return base64.b64decode(result['body'])
        else:
            print(f"Renderer error: {result}")
            return None
    except Exception as e:
        print(f"Renderer invoke error: {e}")
        return None

def get_user_profile(user_id):
    try:
        response = USERS_TABLE.get_item(Key={'user_id': user_id})
        return response.get('Item', {})
    except Exception as e:
        print(f"Profile read error: {e}")
        return {}


def handle_profile_update(sender_id, response):
    """Extract and save profile info from response"""
    try:
        parts = response.split('[PROFILE_UPDATE]')
        if len(parts) < 2:
            return
        json_part = parts[1].strip().replace('```json', '').replace('```', '').strip()
        # Handle case where DOCUMENT_READY comes after
        if '[DOCUMENT_READY]' in json_part:
            json_part = json_part.split('[DOCUMENT_READY]')[0].strip()
        
        profile_data = json.loads(json_part)
        
        # Merge with existing profile
        existing = get_user_profile(sender_id)
        update_expr_parts = ['SET updated_at = :ts']
        expr_values = {':ts': datetime.now(timezone.utc).isoformat()}
        
        for key, value in profile_data.items():
            if value:  # Only update non-empty fields
                safe_key = key.replace('-', '_')
                update_expr_parts.append(f'{safe_key} = :{safe_key}')
                expr_values[f':{safe_key}'] = value
        
        USERS_TABLE.update_item(
            Key={'user_id': sender_id},
            UpdateExpression=', '.join(update_expr_parts),
            ExpressionAttributeValues=expr_values
        )
        print(f"Profile updated for {sender_id}: {list(profile_data.keys())}")
    except Exception as e:
        print(f"Profile update error: {e}")

def send_whatsapp_reply(to, text):
    secrets = get_secrets()
    url = f"https://graph.facebook.com/v21.0/{secrets['phone_number_id']}/messages"

    else:
        _send_whatsapp_text(url, secrets['access_token'], to, text)


def _send_whatsapp_text(url, token, to, text):
    payload = {
        "messaging_product": "whatsapp",
        "recipient_type": "individual",
        "to": to,
        "type": "text",
        "text": {"preview_url": False, "body": text}
    }
    resp = http.request('POST', url, body=json.dumps(payload),
                       headers={'Content-Type': 'application/json', 'Authorization': f'Bearer {token}'})
    print(f"WA send: {resp.status} {resp.data.decode()[:200]}")


def send_whatsapp_document(to, file_bytes, filename):
    """Upload media then send as document"""
    secrets = get_secrets()
    
    # Upload
    upload_url = f"https://graph.facebook.com/v21.0/{secrets['phone_number_id']}/media"
    
    # Determine mime type
    mime_type = 'application/pdf' if filename.endswith('.pdf') else 'text/html'
    
    resp = http.request('POST', upload_url,
        fields={
            'file': (filename, file_bytes, mime_type),
            'messaging_product': 'whatsapp',
            'type': mime_type
        },
        headers={'Authorization': f'Bearer {secrets["access_token"]}'})
    
    result = json.loads(resp.data)
    media_id = result.get('id')
    
    if not media_id:
        print(f"Media upload failed: {result}")
        send_whatsapp_reply(to, "⚠️ File upload failed. Please try again.")
        return
    
    # Send document message
    msg_url = f"https://graph.facebook.com/v21.0/{secrets['phone_number_id']}/messages"
    payload = {
        "messaging_product": "whatsapp",
        "recipient_type": "individual",
        "to": to,
        "type": "document",
        "document": {"id": media_id, "filename": filename}
    }
    resp = http.request('POST', msg_url, body=json.dumps(payload),
                       headers={'Content-Type': 'application/json', 'Authorization': f'Bearer {secrets["access_token"]}'})
    print(f"WA doc send: {resp.status} {resp.data.decode()[:200]}")

```bash
import json
import base64
from io import BytesIO
from weasyprint import HTML

def lambda_handler(event, context):
    html_content = event.get('html', '')
    doc_type = event.get('doc_type', 'document')
    options = event.get('options', {})
    
    if not html_content:
        return {'statusCode': 400, 'body': 'No HTML content'}
    
    try:
        # Convert HTML to PDF using WeasyPrint
        pdf_buffer = BytesIO()
        
        # Add page size CSS
        page_size = options.get('page_size', 'A4')
        margin = options.get('margin', '15mm')
        
        page_css = f"""
        @page {{
            size: {page_size};
            margin: {margin};
        }}
        """
        
        # Inject page CSS into HTML
        styled_html = html_content.replace('</style>', f'{page_css}</style>')
        
        HTML(string=styled_html).write_pdf(pdf_buffer)
        
        pdf_bytes = pdf_buffer.getvalue()
        encoded = base64.b64encode(pdf_bytes).decode('utf-8')
        
        return {
            'statusCode': 200,
            'body': encoded,
            'format': 'pdf',
            'size_bytes': len(pdf_bytes)
        }
    except Exception as e:
        print(f"PDF generation error: {e}")
        # Fallback to HTML
        html_bytes = html_content.encode('utf-8')
        return {
            'statusCode': 200,
            'body': base64.b64encode(html_bytes).decode('utf-8'),
```

```python
def get_domain_knowledge(user_text, history):
    """
    Determine which domain knowledge to retrieve based on user intent.
    Simple keyword/intent matching — upgrade to embeddings later.
    """
    text_lower = user_text.lower()
    
    # Intent patterns
    patterns = {
        'trade_license': ['ট্রেড লাইসেন্স', 'trade license', 'ব্যবসার লাইসেন্স'],
        'birth_certificate': ['জন্ম নিবন্ধন', 'birth certificate', 'জন্ম সনদ'],
        'namjari': ['নামজারি', 'namjari', 'mutation', 'জমির নাম'],
        'passport': ['পাসপোর্ট', 'passport'],
        'tin_certificate': ['টিন', 'tin', 'ট্যাক্স আইডি'],
        'driving_license': ['ড্রাইভিং লাইসেন্স', 'driving license', 'গাড়ির লাইসেন্স'],
        'company_registration': ['কোম্পানি রেজিস্ট্রেশন', 'rjsc', 'company registration'],
        'rent_agreement': ['ভাড়া চুক্তি', 'rent agreement', 'বাসা ভাড়া'],
        'court_case': ['মামলা', 'court', 'মোকদ্দমা', 'lawsuit'],
        'bail': ['জামিন', 'bail'],
        'fir': ['এফআইআর', 'fir', 'জিডি', 'gd', 'general diary'],
    }
    
    for procedure_id, keywords in patterns.items():
        for kw in keywords:
            if kw in text_lower:
                return load_procedure_card(procedure_id)
    
    return None


def load_procedure_card(procedure_id):
    """Load structured knowledge from S3"""
    try:
        response = s3.get_object(
            Bucket='proofsheet-knowledge',
            Key=f'procedures/{procedure_id}.json'
        )
        return json.loads(response['Body'].read())
    except:
        return None
```

```
┌─────────────────────────────────────────────────────┐

                       │
                       ▼

│              LAYER 1: INTENT + CONTEXT               │

│                                                     │

│  • Pull user profile (existing business info)       │

│                                                     │

└──────────────────────┬──────────────────────────────┘
                       │

┌─────────────────────────────────────────────────────┐
```

This isn't three separate API calls. It's **one Bedrock call with structured prompting** that makes the model think in this sequence. The differentiation comes from what you feed it and how you instruct it to reason.

## The Training Pipeline (Not ML Training — Knowledge Ingestion)

### What "Training" Actually Means Here

You're not fine-tuning a model. You're building a **structured knowledge base** that gets injected at inference time. The "training" is:

1. **Curate** raw knowledge (forms, laws, practical advice)
2. **Structure** it into a format the model can reason over
3. **Index** it so the right knowledge surfaces for the right query
4. **Validate** outputs against ground truth

```python
  },

    {

    }

  "confidence": "high|medium|low",
  "needs_review": ["list of items that might be outdated"]
}
"""


def process_raw_knowledge(raw_text, domain_hint=None, source_type=None):
    """
    Take raw text (your notes, a law excerpt, a form description)
    and convert to structured procedure card.
    """
    
    messages = [{"role": "user", "content": f"""
Process this raw knowledge into a structured procedure card.

Source type: {source_type or 'general notes'}
Domain hint: {domain_hint or 'determine from content'}

RAW INPUT:
---
{raw_text}
---

Output ONLY the JSON. No commentary."""}]
    
    response = bedrock.invoke_model(
        modelId='anthropic.claude-sonnet-4-20250514',
        body=json.dumps({
            "anthropic_version": "bedrock-2023-05-31",
            "max_tokens": 4096,
            "system": STRUCTURING_PROMPT,
            "messages": messages
        })
    )
    
    result = json.loads(response['body'].read())
    structured = result['content'][0]['text']
    
    # Parse and validate
    try:
        card = json.loads(structured.replace('```json', '').replace('```', ''))
        validate_card(card)
        return card
    except (json.JSONDecodeError, ValidationError) as e:
        return {"error": str(e), "raw_output": structured}


def validate_card(card):
    """Ensure all required fields exist and make sense"""
    required = ['id', 'domain', 'title_bn', 'required_documents', 'process_steps']
    for field in required:
        if field not in card:
            raise ValidationError(f"Missing required field: {field}")
    
    if not card.get('pro_tips'):
        card['pro_tips'] = ["[NEEDS PRO TIPS — practical wisdom not yet captured]"]
        card.setdefault('needs_review', []).append('pro_tips missing')
```

## Bengali Processing: Not Just Translation

### The Language Problem

Bengali in Bangladesh isn't one language. It's a spectrum:

```
Formal Written (সাধু/চলিত)  ←→  Spoken Standard  ←→  Regional  ←→  Banglish
"আবেদনকারীর নিকট"          "আবেদনকারীর কাছে"    "আবেদন    "applicant er
                                                    korta r    kache"
                                                    kase"
```

Your users write across this entire spectrum. The model needs to:

1. **Understand** any point on the spectrum
2. **Reply** at the user's level (mirror their register)
3. **Generate documents** at the appropriate formality for the document type
4. **Code-switch** naturally (not mechanically)

### The Approach: Register Detection + Matched Response

```python
# This goes in the system prompt as behavioral instruction
```

### Bengali-Specific Processing Challenges

**Problem 1: Multiple scripts in one message**

Users write Bangla in both:
- Unicode Bengali script: "আমি একটা invoice চাই"
- Roman transliteration: "ami ekta invoice chai"

The model handles this natively, but for **search and intent matching**, you need both:

```python
# knowledge_retrieval.py

# Map common Roman-Bangla to Bengali for search
TRANSLITERATION_MAP = {
    'namjari': 'নামজারি',
    'trade license': 'ট্রেড লাইসেন্স',
    'jomi': 'জমি',
    'dolil': 'দলিল',
    'mamla': 'মামলা',
    'thana': 'থানা',
    'gd': 'জিডি',
    'nid': 'জাতীয় পরিচয়পত্র',
    'tin': 'টিআইএন',
    'khajna': 'খাজনা',
    'dcr': 'ভূমি উন্নয়ন কর',
    'porcha': 'পর্চা',
    'khotian': 'খতিয়ান',
    'mouza': 'মৌজা',
    'warrant': 'ওয়ারেন্ট',
    'jamin': 'জামিন',
    'dorkhasto': 'দরখাস্ত',
    'baina': 'বায়না',
    'registry': 'রেজিস্ট্রি',
    'rin': 'ঋণ',
    'bank loan': 'ব্যাংক ঋণ',
    'bkash': 'বিকাশ',
    'nagad': 'নগদ',
    'challan': 'চালান',
}

def normalize_for_search(text):
    """Normalize user input for knowledge retrieval"""
    text_lower = text.lower()
    
    # Check Roman transliterations
    matched_terms = []
    for roman, bangla in TRANSLITERATION_MAP.items():
        if roman in text_lower:
            matched_terms.append(bangla)
            matched_terms.append(roman)
    
    # Return both original text and matched terms for search
    return {
        'original': text,
        'normalized': text_lower,
        'matched_terms': matched_terms
    }
```

**Problem 2: Document output in proper Bengali typography**

Government offices expect specific formatting:
- Date format: ২৫ জুন ২০২৫ (not 25 June 2025) in Bangla documents
- Numbers: Bengali digits (১২৩) in formal docs, English digits (123) in business docs
- Honorifics: জনাব/জনাবা, মহোদয়/মহোদয়া — gender-appropriate
- Address format: Bangladeshi convention (village/road → post office → upazila → district)

```python
# bengali_utils.py

BANGLA_DIGITS = {'0': '০', '1': '১', '2': '২', '3': '৩', '4': '৪',
                 '5': '৫', '6': '৬', '7': '৭', '8': '৮', '9': '৯'}

BANGLA_MONTHS = {
    'January': 'জানুয়ারি', 'February': 'ফেব্রুয়ারি', 'March': 'মার্চ',
    'April': 'এপ্রিল', 'May': 'মে', 'June': 'জুন',
    'July': 'জুলাই', 'August': 'আগস্ট', 'September': 'সেপ্টেম্বর',
    'October': 'অক্টোবর', 'November': 'নভেম্বর', 'December': 'ডিসেম্বর'
}

def to_bangla_digits(number_str):
    """Convert 12345 to ১২৩৪৫"""
    return ''.join(BANGLA_DIGITS.get(c, c) for c in str(number_str))

def to_bangla_date(date_str, formal=True):
    """Convert '2025-06-25' to '২৫ জুন ২০২৫' or '25 June 2025'"""
    from datetime import datetime
    dt = datetime.strptime(date_str, '%Y-%m-%d')
    
    if formal:
        day = to_bangla_digits(dt.day)
        month = BANGLA_MONTHS[dt.strftime('%B')]
        year = to_bangla_digits(dt.year)
        return f"{day} {month} {year}"
    else:
        return dt.strftime('%d %B %Y')

def format_amount_bangla(amount, formal=False):
    """Format currency amount"""
    if formal:
        return f"৳{to_bangla_digits(f'{amount:,.0f}')}"
    else:
        return f"৳{amount:,.0f}"

def get_honorific(name, gender=None, context='formal'):
    """
    Determine appropriate Bengali honorific.
    In formal documents: জনাব (male), জনাবা (female)
    In letters: মহোদয় (male), মহোদয়া (female)
    If gender unknown: মহোদয়/মহোদয়া or just the name
    """
    if context == 'application':
        if gender == 'male': return f"জনাব {name}"
        elif gender == 'female': return f"জনাবা {name}"
        else: return name
    elif context == 'letter_salutation':
        if gender == 'male': return "মহোদয়"
        elif gender == 'female': return "মহোদয়া"
        else: return "মহোদয়/মহোদয়া"
    return name
```

## The Reasoning Engine: How It Decides What To Do

This is the part that prevents it from being "Perplexity with a printer."

### The Decision Framework (Embedded in System Prompt)

```python
REASONING_FRAMEWORK = """
## HOW YOU THINK (Internal reasoning before every response):

### Step 1: CLASSIFY the interaction type
- INFORMATION: User wants to know something ("fee কত?")
- PROCEDURE: User wants to DO something ("কিভাবে করব?")
- DOCUMENT: User wants you to PRODUCE something ("বানাও")
- EDIT: User wants to MODIFY a previous output
- CLARIFICATION: User is confused about something
- STATUS: User is mid-process and updating you

### Step 2: ASSESS what they know vs don't know
- If they ask about fees → they probably don't know the full process
- If they mention a specific office → they know the basics
- If they use official terminology → they've done research
- If they describe a problem → they might not know the right solution

### Step 3: DECIDE response depth
- New to the topic: Give overview + immediate next step + offer document
- Mid-process: Address their specific point + flag any upcoming gotcha
- Experienced but stuck: Focus on the specific blocker only
- Ready for document: Confirm → Generate → Guide on what to do with it

### Step 4: ADD VALUE beyond the ask
- ALWAYS mention the one thing they probably don't know
- ALWAYS suggest the document you can produce if relevant
- NEVER lecture — one additional insight per response, max
- If there's a deadline or time-sensitive element, flag it FIRST

### Step 5: GUIDE forward
- End every substantive response with a clear next action
- Either: "want me to make the document?" 
- Or: "do X next, then come back and I'll do Y"
- Or: "tell me [specific detail] and I'll fill this out for you"

## WHAT MAKES YOU DIFFERENT FROM A SEARCH ENGINE:
- You know the PRACTICAL process, not just the official one
- You know what goes wrong and how to prevent it
- You produce the actual document, filled out correctly
- You remember what the user already told you
- You guide them through multi-step processes across sessions
"""
```

```python
def retrieve_knowledge(user_text, history):
    """
    Multi-strategy knowledge retrieval.
    Phase 1: keyword matching + procedure card lookup
    Phase 2: Bedrock Knowledge Base (vector embeddings)
    """

    procedure = match_procedure(user_text)
    if procedure:

    return None
```

## How Knowledge Gets Into the Response

The key architectural decision: **knowledge is injected into the system prompt dynamically, not hard-coded.**

```python
def build_system_prompt(user_text, history, user_profile, last_doc):
    """
    Assemble the full system prompt with all context.
    This is what makes the model intelligent about the specific query.
    """
    
    # Base personality + rules (always present)
    prompt = CORE_PERSONALITY + LANGUAGE_RULES + REASONING_FRAMEWORK
    
    # User profile (if known)
    prompt += build_profile_section(user_profile)
    
    # Retrieved knowledge (query-specific)
    knowledge = retrieve_knowledge(user_text, history)
    if knowledge:
        prompt += build_knowledge_section(knowledge)
    
    # Last document context (for edits)
    if last_doc:
        prompt += build_last_doc_section(last_doc)
    
    # Active procedure context (multi-turn tracking)
    active = get_active_procedure(history)
    if active:
        prompt += build_procedure_progress_section(active, history)
    
    # Today's date and contextual info
    prompt += build_context_section()
    
    return prompt


def build_knowledge_section(knowledge):
    """Format retrieved knowledge for the model"""
    
    if knowledge['type'] == 'procedure_card':
        card = knowledge['data']
        section = f"""
## 📋 ACTIVE KNOWLEDGE: {card['title_bn']} ({card['title_en']})

**Authority:** {card['authority']}
**Jurisdiction:** {', '.join(card['jurisdiction'])}

### Required Documents:
{format_documents_list(card['required_documents'])}

### Fees:
{format_fees(card['fees'])}

### Process Steps:
{format_steps(card['process_steps'])}

### Timeline:
- Official: {card['timeline']['official']}
- Realistic: {card['timeline']['realistic']}
- Factors: {card['timeline']['factors']}

### Common Rejections:
{format_rejections(card['common_rejections'])}

### Pro Tips (SHARE THESE):
{chr(10).join(f'💡 {tip}' for tip in card['pro_tips'])}

### Related: {', '.join(card.get('related_procedures', []))}

USE THIS KNOWLEDGE naturally in your response. Don't dump it all at once.
Share what's relevant to the user's SPECIFIC situation.
If they're asking generally → give overview + offer to help with specific steps.
If they're at a specific step → focus on THAT step + the next gotcha.
"""
        return section
    
    elif knowledge['type'] == 'domain_overview':
        return f"\n## Domain Context:\n{knowledge['data']}\n"
    
    return ""
```

```python
def handle_user_correction(sender_id, correction_text, procedure_id):
    """
    When a user says "this is wrong" or provides updated info.
    Don't auto-update — flag for human review.
    """
    
    correction_record = {
        'procedure_id': procedure_id,
        'reported_by': sender_id,
        'correction': correction_text,
        'timestamp': datetime.now(timezone.utc).isoformat(),
        'status': 'pending_review'
    }
    
    # Save to corrections table
    corrections_table.put_item(Item=correction_record)
    
    # Notify you (via SNS or just a log you check)
    print(f"CORRECTION REPORTED: {procedure_id} — {correction_text}")
    
    return "ধন্যবাদ! আপনার feedback record করা হয়েছে। আমরা verify করে update করব। 🙏"


def refresh_knowledge_card(procedure_id, updates):
    """
    Apply verified updates to a procedure card.
    Called by you (the human) after reviewing corrections.
    """
    card = load_card(procedure_id)
    
    for field, new_value in updates.items():
        if '.' in field:  # Nested field like "fees.total_typical"
            keys = field.split('.')
            obj = card
            for key in keys[:-1]:
                obj = obj[key]
            obj[keys[-1]] = new_value
        else:
            card[field] = new_value
    
    card['last_verified'] = datetime.now(timezone.utc).strftime('%Y-%m-%d')
    
    # Save back to S3
    s3.put_object(
        Bucket=KNOWLEDGE_BUCKET,
        Key=f'procedures/{procedure_id}.json',
        Body=json.dumps(card, ensure_ascii=False, indent=2),
        ContentType='application/json'
    )
    
    # Clear any cache
    invalidate_cache(procedure_id)
```

def lambda_handler(event, context):
    print(f"Event: {json.dumps(event)}")
    method = event.get('requestContext', {}).get('http', {}).get('method', 'POST')

    if method == 'GET':
        return handle_verification(event)

    try:
        body = json.loads(event.get('body', '{}'))
    except json.JSONDecodeError:
        return {'statusCode': 200}

    obj = body.get('object', '')
    if obj == 'whatsapp_business_account':
        handle_whatsapp(body)
    elif obj == 'page':
        handle_messenger(body)

    return {'statusCode': 200}


def handle_verification(event):
    params = event.get('queryStringParameters', {}) or {}
    mode = params.get('hub.mode', '')
    token = params.get('hub.verify_token', '')
    challenge = params.get('hub.challenge', '')
    secrets = get_secrets()
    if mode == 'subscribe' and token == secrets['verify_token']:
        return {'statusCode': 200, 'headers': {'Content-Type': 'text/plain'}, 'body': challenge}
    return {'statusCode': 403, 'body': 'Forbidden'}

def handle_whatsapp(body):
    for entry in body.get('entry', []):
        for change in entry.get('changes', []):
            value = change.get('value', {})
            if 'messages' not in value:
                continue
            for message in value['messages']:
                sender = message['from']
                msg_type = message['type']
                if msg_type == 'text':
                    user_text = message['text']['body']
                elif msg_type == 'interactive':
                    interactive = message.get('interactive', {})
                    user_text = (interactive.get('button_reply', {}).get('title') or
                                interactive.get('list_reply', {}).get('title') or '[interactive]')
                else:
                    send_whatsapp_reply(sender, "এখন শুধু text support করি। Voice ও image শীঘ্রই আসছে! 🙏")
                    return
                process_and_respond(sender, user_text, 'whatsapp')


def handle_messenger(body):
    for entry in body.get('entry', []):
        for messaging_event in entry.get('messaging', []):
            if messaging_event.get('message', {}).get('is_echo'):
                continue
            sender_id = messaging_event['sender']['id']
            if 'message' in messaging_event:
                message = messaging_event['message']
                if 'text' in message:
                    user_text = message['text']
                elif 'attachments' in message:
                    send_messenger_reply(sender_id, "এখন শুধু text support করি। 🙏")
                    return
                else:
                    return
                process_and_respond(sender_id, user_text, 'messenger')
            elif 'postback' in messaging_event:
                payload = messaging_event['postback'].get('title', messaging_event['postback'].get('payload', ''))
                process_and_respond(sender_id, payload, 'messenger')

def retrieve_knowledge(user_text, history):
    """Retrieve relevant procedure card based on user text and conversation history"""
    text_lower = user_text.lower()
    text_combined = user_text  # Keep original for Bangla matching
    
    # Also check recent history for context
    recent_context = ' '.join([m['content'] for m in history[-4:]]) if history else ''
    search_text = f"{text_combined} {recent_context}".lower()
    
    best_match = None
    best_score = 0
    
    for proc_id, keywords in PROCEDURE_KEYWORDS.items():
        score = 0
        for kw in keywords:
            kw_lower = kw.lower()
            if kw_lower in text_lower or kw in text_combined:
                score += 3  # Direct match in current message
            elif kw_lower in search_text:
                score += 1  # Match in context
        
        if score > best_score:
            best_score = score
            best_match = proc_id
    
    if best_match and best_score >= 2:
        return load_knowledge_card(best_match)
    
    return None


def load_knowledge_card(procedure_id):
    """Load procedure card from S3 (with caching)"""
    global _knowledge_cache
    
    if procedure_id in _knowledge_cache:
        return _knowledge_cache[procedure_id]
    
    try:
        response = s3.get_object(
            Bucket=KNOWLEDGE_BUCKET,
            Key=f'procedures/{procedure_id}.json'
        )
        card = json.loads(response['Body'].read())
        _knowledge_cache[procedure_id] = card
        return card
    except Exception as e:
        print(f"Knowledge load error ({procedure_id}): {e}")
        return None

def call_bedrock(user_text, history, system_prompt):
    messages = []
    for msg in history[-14:]:
        messages.append({"role": msg['role'], "content": msg['content']})
    messages.append({"role": "user", "content": user_text})

    try:
        response = bedrock.invoke_model(
            modelId='anthropic.claude-sonnet-4-20250514',
            body=json.dumps({
                "anthropic_version": "bedrock-2023-05-31",
                "max_tokens": 4096,
                "system": system_prompt,
                "messages": messages
            })
        )
        result = json.loads(response['body'].read())
        return result['content'][0]['text']
    except Exception as e:
        print(f"Bedrock error: {e}")
        return "দুঃখিত, একটু সমস্যা হয়েছে। আবার চেষ্টা করুন। 🙏"

def handle_document_generation(sender_id, response, channel):
    parts = response.split('[DOCUMENT_READY]')
    conversational_reply = parts[0].strip()
    json_part = parts[1]
    if '[PROFILE_UPDATE]' in json_part:
        json_part = json_part.split('[PROFILE_UPDATE]')[0]
    json_part = json_part.strip().replace('```json', '').replace('```', '').strip()

    try:
        doc_data = json.loads(json_part)
    except json.JSONDecodeError as e:
        print(f"JSON error: {e}\nRaw: {json_part[:500]}")
        send_response(channel, sender_id, conversational_reply + "\n\n⚠️ Document generation এ সমস্যা হয়েছে। আবার চেষ্টা করুন।")
        return

    doc_type = doc_data.get('type', 'unknown')
    doc_content = doc_data.get('data', {})
    
    send_response(channel, sender_id, conversational_reply)
    
    # Render HTML
    html_content = render_document(doc_type, doc_content)
    
    # Try renderer Lambda, fallback to HTML
    pdf_bytes = None
    try:
        resp = lambda_client.invoke(
            FunctionName='proofsheet-renderer',
            InvocationType='RequestResponse',
            Payload=json.dumps({'html': html_content, 'doc_type': doc_type})
        )
        result = json.loads(resp['Payload'].read())
        if result.get('statusCode') == 200:
            import base64
            pdf_bytes = base64.b64decode(result['body'])
    except Exception as e:
        print(f"Renderer error: {e}")
    
    if pdf_bytes:
        file_bytes = pdf_bytes
        ext = 'html'  # Still HTML from basic renderer
    else:
        file_bytes = html_content.encode('utf-8')
        ext = 'html'
    
    timestamp_str = datetime.now(timezone.utc).strftime('%Y%m%d_%H%M%S')
    filename = f"{doc_type}_{timestamp_str}.{ext}"
    
    # Save to S3
    try:
        s3.put_object(Bucket=DOCS_BUCKET, Key=f"{sender_id}/{filename}", Body=file_bytes, ContentType='text/html')
    except: pass
    
    # Send file
    send_file_response(channel, sender_id, file_bytes, filename)
    
    # Save doc record
    save_document_record(sender_id, doc_type, doc_content)

def get_user_profile(user_id):

        resp = USERS_TABLE.get_item(Key={'user_id': user_id})

def handle_profile_update(sender_id, response):
    try:
        parts = response.split('[PROFILE_UPDATE]')
        if len(parts) < 2: return
        json_part = parts[-1].strip().replace('```json','').replace('```','').strip()
        if '[DOCUMENT_READY]' in json_part:
            json_part = json_part.split('[DOCUMENT_READY]')[0].strip()
        profile_data = json.loads(json_part)
        
        update_parts = ['SET updated_at = :ts']
        expr_vals = {':ts': datetime.now(timezone.utc).isoformat()}
        for k, v in profile_data.items():
            if v:
                update_parts.append(f'{k} = :{k}')
                expr_vals[f':{k}'] = v
        
        USERS_TABLE.update_item(
            Key={'user_id': sender_id},
            UpdateExpression=', '.join(update_parts),
            ExpressionAttributeValues=expr_vals
        )
        print(f"Profile updated: {sender_id}")
    except Exception as e:
        print(f"Profile update error: {e}")

def get_last_document(user_id):
    try:
        from boto3.dynamodb.conditions import Key
        resp = DOCUMENTS_TABLE.query(
            KeyConditionExpression=Key('user_id').eq(user_id),
            ScanIndexForward=False, Limit=1
        )
        items = resp.get('Items', [])
        return items[0] if items else None
    except: return None


def save_document_record(user_id, doc_type, doc_data):
    try:
        DOCUMENTS_TABLE.put_item(Item={
            'user_id': user_id,
            'created_at': datetime.now(timezone.utc).isoformat(),
            'doc_type': doc_type,
            'doc_data': json.loads(json.dumps(doc_data, default=str))
        })
    except Exception as e:
        print(f"Doc save error: {e}")

def get_conversation_history(sender_id):
    try:
        from boto3.dynamodb.conditions import Key
        resp = CONVERSATION_TABLE.query(
            KeyConditionExpression=Key('phone_number').eq(sender_id),
            ScanIndexForward=False, Limit=20
        )
        items = resp.get('Items', [])
        items.reverse()
        return [{'role': item['role'], 'content': item['content']} for item in items]
    except: return []

            'role': role, 'content': content[:5000]
        })

def send_response(channel, recipient, text):

def send_whatsapp_reply(to, text):
    secrets = get_secrets()
    url = f"https://graph.facebook.com/v21.0/{secrets['phone_number_id']}/messages"
    if len(text) > 4000:
        for chunk in [text[i:i+4000] for i in range(0, len(text), 4000)]:
            _send_wa(url, secrets['access_token'], to, chunk)
    else:
        _send_wa(url, secrets['access_token'], to, text)

def _send_wa(url, token, to, text):
    resp = http.request('POST', url, body=json.dumps({"messaging_product":"whatsapp","to":to,"type":"text","text":{"body":text}}),
                       headers={'Content-Type':'application/json','Authorization':f'Bearer {token}'})
    print(f"WA: {resp.status}")

def send_whatsapp_document(to, file_bytes, filename):
    secrets = get_secrets()
    upload_url = f"https://graph.facebook.com/v21.0/{secrets['phone_number_id']}/media"
    mime = 'application/pdf' if filename.endswith('.pdf') else 'text/html'
    resp = http.request('POST', upload_url, fields={'file':(filename,file_bytes,mime),'messaging_product':'whatsapp','type':mime},
                       headers={'Authorization':f'Bearer {secrets["access_token"]}'})
    media_id = json.loads(resp.data).get('id')
    if media_id:
        msg_url = f"https://graph.facebook.com/v21.0/{secrets['phone_number_id']}/messages"
        http.request('POST', msg_url, body=json.dumps({"messaging_product":"whatsapp","to":to,"type":"document","document":{"id":media_id,"filename":filename}}),
                    headers={'Content-Type':'application/json','Authorization':f'Bearer {secrets["access_token"]}'})

def send_messenger_reply(recipient_id, text):
    secrets = get_secrets()
    token = secrets.get('page_access_token','')
    if not token: return
    url = f"https://graph.facebook.com/v21.0/me/messages?access_token={token}"
    if len(text) > 2000:
        for chunk in [text[i:i+2000] for i in range(0, len(text), 2000)]:
            _send_msg(url, recipient_id, chunk)
    else:
        _send_msg(url, recipient_id, text)

def _send_msg(url, recipient_id, text):
    resp = http.request('POST', url, body=json.dumps({"recipient":{"id":recipient_id},"message":{"text":text},"messaging_type":"RESPONSE"}),
                       headers={'Content-Type':'application/json'})
    print(f"MSG: {resp.status}")

def send_messenger_document(recipient_id, file_bytes, filename):
    secrets = get_secrets()
    token = secrets.get('page_access_token','')
    if not token: return
    url = f"https://graph.facebook.com/v21.0/me/messages?access_token={token}"
    mime = 'application/pdf' if filename.endswith('.pdf') else 'text/html'
    resp = http.request('POST', url, fields={
        'recipient':json.dumps({"id":recipient_id}),
        'message':json.dumps({"attachment":{"type":"file","payload":{"is_reusable":False}}}),
        'filedata':(filename,file_bytes,mime),'messaging_type':'RESPONSE'})
    print(f"MSG doc: {resp.status}")
